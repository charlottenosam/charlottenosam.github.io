# Get spectra within PSF aperture, where sky lines are masked
spec3D = cubehighz.dat3D_masked * cubehighz.cube.mask_psf / cubehighz.scale * 1E19
err3D  = cubehighz.err3D * cubehighz.cube.mask_psf / cubehighz.scale * 1E19

line = np.nansum(spec3D, axis=(1, 2))
err  = np.sqrt(np.nansum(err3D**2., axis=(1, 2)))
# Block out bad telluric region
line[(cube_obj.wavevec < 12900)*(cube_obj.wavevec > 12500)] = np.nan
line[line == 0.] = np.nan

# Plot spextrum
plt.figure(figsize=(12,4))

plt.axhline(y=0., c='k', lw=1)
plt.axvline(x=1.3E4, c='k', lw=1)
# plt.scatter(cube_obj.wavevec, line, alpha=0.5, c='0.8')
# plt.fill_between(cube_obj.wavevec, 0., err, color='0.8')

# Boxcar weighted smooth
px_smooth = 20
clip_px = px_smooth/2
no_edge = np.where(err > 0.)
line_boxcar = KT.smooth_boxcar(np.nan_to_num(line[no_edge]), px_smooth, weights=1./err[no_edge], squared=True)
line_boxcar[line_boxcar == 0.] = np.nan
plt.plot(cube_obj.wavevec[no_edge][clip_px:-clip_px+(1 - px_smooth % 2)], line_boxcar,
        drawstyle='steps-mid')

# Model ~ estimated from draft
mod = np.zeros_like(line)
mod[cube_obj.wavevec > 13000] = 0.2
plt.plot(cube_obj.wavevec, mod, c='r')

plt.xlim(1.02E4, 1.35E4)
plt.ylim(-2,5)
plt.xlabel('Wavelength $\AA$')
plt.ylabel(r'$f_{\lambda} \times 10^{-19}$ cgs')
